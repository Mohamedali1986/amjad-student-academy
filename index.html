function doPost(e) { try { const data = JSON.parse(e.postData.contents); const sheet = getOrCreateSheet(); const row = [ new Date(), data["اسم الطالب"] || "", data["العمر"] || "", data["رقم تليفون الطالب"] || "", data["النوع"] || "", data["مقدار الحفظ ومستواه"] || "", data["المعرفة بالتجويد ومستواها"] || "", data["هل يعاني من صعوبات تعلم"] || "", data["تفاصيل صعوبات التعلم"] || "", data["اسم ولي الأمر"] || "", data["تليفون ولي الأمر"] || "", data["صلة القرابة"] || "", data["الدورة المختارة"] || "", data["نظام الحصة"] || "", data["فترة الحفظ"] || "", data["اختيار المعلم"] || "", data["عدد الحلقات"] || "", data["الأيام المختارة"] || "", data["الوقت المفضل"] || "" ];
